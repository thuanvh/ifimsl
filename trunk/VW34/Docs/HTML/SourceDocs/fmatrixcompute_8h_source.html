00125         const VW::Normalisation& n0,
00126         const VW::Normalisation& n1)
00127 {
00128   VW::FMatrix result;
00129   const double s0 = n0.GetScale(), s1 = n1.GetScale();
00130   const double x0 = n0.GetCentreX(), x1 = n1.GetCentreX();
00131   const double y0 = n0.GetCentreY(), y1 = n1.GetCentreY();
00132   const double s01 = s0*s1;

00134   result[0][0] = s01*F[0][0];

00137 
00138   result[1][0] = s01*F[1][0];
00139   result[1][1] = s01*F[1][1];

00141  
00142   result[2][0] = s0 *F[2][0] - x1*result[0][0] - y1*result[1][0];
00143   result[2][1] = s0 *F[2][1] - x1*result[0][1] - y1*result[1][1];
00144   result[2][2] = F[2][2] - s1*x1*F[0][2] - s1*y1*F[1][2] - x0*result[2][0] - y0*result[2][1];
